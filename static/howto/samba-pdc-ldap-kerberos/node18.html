Configurazione Kerberos

Configuriamo ora il client kerberos editando /etc/krb5.conf: 
[libdefaults]
        default_realm = ESEMPIO.LAN
        ticket_lifetime = 80000
        kdc_timesync = 1
        ccache_type = 4
        forwardable = true
        proxiable = true

        v4_instance_resolve = false
        v4_name_convert = {
                host = {
                        rcmd = host
                        ftp = ftp
                }
                plain = {
                        something = something-else
                }
        }

[realms]
ESEMPIO.LAN = {
         kdc = pdc.esempio.lan 
         admin_server = pdc.esempio.lan 
         default_domain = esempio.lan
}

[domain_realm]
        .esempio.lan  = ESEMPIO.LAN
        esempio.lan  = ESEMPIO.LAN
