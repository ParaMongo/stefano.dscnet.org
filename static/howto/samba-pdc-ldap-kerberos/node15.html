Installazione dei pacchetti necessari

Passiamo all'installazione dei pacchetti: 
# DEBIAN_FRONTEND=noninteractive
# export DEBIAN_FRONTEND=noninteractive
# apt-get update
# apt-get install -y heimdal-clients krb5-config libnss-ldap libsasl2-modules-gssapi-heimdal \
ldap-utils libpam-krb5 libpam-ldap cupsys-common libcupsys2 keyutils winbind smbclient samba \
samba-common smbfs libsmbclient ipcalc dnsutils ntp
# unset DEBIAN_PRIORITY
# unset DEBIAN_FRONTEND

fermiamo ora i servizi appena installati: 

# /etc/init.d/samba stop
# /etc/init.d/winbind stop
# /etc/init.d/ntp stop
